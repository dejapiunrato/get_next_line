char	*get_next_line(int fd)
{
	static char	*buf;
	char		*line;
	char		*new_line;

	if (fd < 0 || BUFFER_SIZE <= 0)
		return (NULL);

	// Leer el contenido del archivo
	buf = ft_new_line(fd, buf);
	if (!buf)
		return (NULL);

	// Buscar el salto de línea en el buffer
	new_line = ft_strenter(buf);

	// Si encontramos un salto de línea
	if (new_line)
	{
		*new_line = '\0'; // Terminar la línea en el salto de línea
		line = ft_strdup(buf); // Duplicar la línea encontrada
		if (!line)
			return (NULL);
		buf = ft_strdup(new_line + 1); // Mover el buffer estático
		return (line); // Retornar la línea
	}
	
	// Si no hay salto de línea pero aún hay contenido en el buffer
	if (*buf)
	{
		line = ft_strdup(buf); // Duplicar el contenido restante
		free(buf); // Liberar el buffer estático
		buf = NULL;
		return (line); // Retornar la última línea
	}

	return (NULL); // En caso de error o si no queda nada por leer
}
